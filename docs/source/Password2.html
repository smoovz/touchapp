Ext.define('Smoovz.form.Password', {
    extend: 'Ext.form.Panel',
    alias: 'widget.passwordform',
    
     requires: [
        'Ext.Button',
        'Ext.field.Email',
        'Ext.form.FieldSet',
        'Ext.TitleBar',
        'Smoovz.util.Il8n'
    ],
    
    config: {
        layout: {
            type: 'vbox',
            align: 'center'
        },
        items: [{
            xtype: 'titlebar',
            docked: 'top'
        }, {
            xtype: 'fieldset',
            width: '90%',
            items: [
                {
                    name: 'email',
                    xtype: 'emailfield'
                }
            ]
        },{
            xtype: 'button',
            ui: 'action',
            itemId: 'newPasswordBtn'
        }]
    },
    
    initialize: function () {
        var me = this;
        
        me.callParent();
        
        me.setUrl(Config.getApiUrl() + 'password');
        
        me.down('titlebar')
            .setTitle(Il8n.translate('new_password_title'));
        me.down('fieldset')
            .setTitle(Il8n.translate('new_password_text'));
        me.down('emailfield')
            .setLabel(Il8n.translate('email'))
            .setPlaceHolder(Il8n.translate('email_placeholder'));
        me.down('button')
            .setText(Il8n.translate('new_password_btn'));
    }
});
